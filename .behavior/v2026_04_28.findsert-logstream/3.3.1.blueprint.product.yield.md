## summary

add optional `outlets` parameter to `generateLogMethods` that enables log events to flow to external destinations (e.g., CloudWatch) in addition to console. ship `genCloudwatchOutlet` as the first outlet implementation with lambda-style log group name defaults via `asLambdaStyleLogGroupName`.

**scope:**
- extend `generateLogMethods` contract: `{ level: { minimum }, outlets?: [] }`
- add `LogOutlet` interface for custom outlets
- add `LogEvent` interface for outlet events
- add `genCloudwatchOutlet` for CloudWatch destination
- add `asLambdaStyleLogGroupName` transformer for log group name format
- auto-flush on interval (30s) and process exit
- backwards compatible: no outlets = console only (current behavior)

### generateLogMethods (extended)

```typescript
export const generateLogMethods = ({
  level,
  outlets,
}: {
  level?: {
    minimum?: LogLevel;
  };
  outlets?: LogOutlet[];
} = {}): LogMethods => {
  const minimalLogLevel = level?.minimum ?? getRecommendedMinimalLogLevelForEnvironment();
  return {
    error: generateLogMethod({ level: LogLevel.ERROR, minimalLogLevel, outlets }),
    warn: generateLogMethod({ level: LogLevel.WARN, minimalLogLevel, outlets }),
    info: generateLogMethod({ level: LogLevel.INFO, minimalLogLevel, outlets }),
    debug: generateLogMethod({ level: LogLevel.DEBUG, minimalLogLevel, outlets }),
  };
};
```

### generateLogMethod (extended)

```typescript
export const generateLogMethod = ({
  level,
  minimalLogLevel,
  outlets,
}: {
  level: LogLevel;
  minimalLogLevel: LogLevel;
  outlets?: LogOutlet[];
}) => {
  return (message: string, metadata?: object) => {
    // check level threshold
    if (!aIsEqualOrMoreImportantThanB({ a: level, b: minimalLogLevel })) return;

    // output to console (current behavior)
    const consoleMethod = aIsEqualOrMoreImportantThanB({ a: level, b: LogLevel.WARN })
      ? console.warn
      : console.log;
    consoleMethod(formatLogContentsForEnvironment({ level, timestamp: new Date().toISOString(), message, metadata }));

    // dispatch to outlets
    if (outlets) {
      const event: LogEvent = { level, timestamp: new Date().toISOString(), message, metadata };
      for (const outlet of outlets) {
        try {
          outlet.send(event);
        } catch (error) {
          console.warn('outlet.send failed', { error: error instanceof Error ? error.message : error });
        }
      }
    }
  };
};
```

### genCloudwatchOutlet

```typescript
export const genCloudwatchOutlet = ({
  logGroup,
  logStream,
  skipLogGroupCreation,
  flushInterval,
  maxBufferSize,
}: {
  logGroup?: string;
  logStream?: string;
  skipLogGroupCreation?: boolean;
  flushInterval?: number;      // default: 5_000 (5s)
  maxBufferSize?: number;      // default: 10MB, force-flush when exceeded
} = {}): LogOutlet => {
  // validate AWS credentials at creation
  if (!process.env.AWS_REGION && !process.env.AWS_DEFAULT_REGION)
    throw new Error('AWS_REGION or AWS_DEFAULT_REGION required for cloudwatch outlet');

  // derive defaults
  const logGroupName = logGroup ?? asLambdaStyleLogGroupName();
  const logStreamName = logStream ?? `${new Date().toISOString().split('T')[0]}/${uuid()}`;

  // create client
  const client = new CloudWatchLogsClient({ region: process.env.AWS_REGION ?? process.env.AWS_DEFAULT_REGION });

  // buffer for events
  const buffer: LogEvent[] = [];
  let initialized = false;

  // setup auto-flush (configurable interval + process exit hooks)
  const intervalMs = flushInterval ?? 5_000;
  const bufferLimit = maxBufferSize ?? 10_000_000; // 10MB default
  const flushTimer = setInterval(() => flush(), intervalMs);
  process.on('beforeExit', () => flush());
  process.on('SIGTERM', () => flush());
  process.on('SIGINT', () => flush());

  const init = async () => {
    if (initialized) return;
    if (!skipLogGroupCreation) await findsertLogGroup(client, logGroupName);
    await findsertLogStream(client, logGroupName, logStreamName);
    initialized = true;
  };

  const flush = async () => {
    if (buffer.length === 0) return;
    await init();
    // batch events (max 1MB, 10k events) — per [R.EXT] [3][10]
    const events = buffer.splice(0, buffer.length);
    await client.send(new PutLogEventsCommand({
      logGroupName,
      logStreamName,
      logEvents: events.map(e => ({ timestamp: Date.parse(e.timestamp), message: JSON.stringify(e) })),
    }));
  };

  return {
    send: (event: LogEvent) => {
      buffer.push(event);
      // force-flush if buffer exceeds size limit
      const bufferSize = JSON.stringify(buffer).length;
      if (bufferSize > bufferLimit) flush();
    },
    flush,
  };
};
```

### asLambdaStyleLogGroupName

```typescript
export const asLambdaStyleLogGroupName = ({
  service,
  env,
}: {
  service?: string;
  env?: { access?: string };
} = {}): string => {
  const serviceName = service ?? getPackageJsonName();
  const envAccess = env?.access ?? getEnvironmentAccess();
  return `/aws/lambda/${serviceName}-${envAccess}`;
};
```

## risks

| risk | mitigation |
|------|------------|
| AWS SDK size | @aws-sdk/client-cloudwatch-logs is tree-shakeable |
| process exit race | flush returns Promise; use process.exitCode instead of process.exit |
| outlet error crashes app | try/catch around outlet.send, warn to console |
| buffer memory | maxBufferSize option with 10MB default, force-flush when exceeded |
| batch send fails | v1 does not retry; logs lost on transient failure; users who require guaranteed delivery should implement retry wrapper |
